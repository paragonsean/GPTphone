125 # Queue for incoming WebSocket messages
126 message_queue = asyncio.Queue()

128 async def websocket_listener():
129 try:
130 while True:
131 data = await websocket.receive_text()
132 await message_queue.put(json.loads(data))
133 except WebSocketDisconnect:
134 logger.info("WebSocket disconnected")

136 async def message_processor():
137 while True:
138 msg = await message_queue.get()
139 if msg['event'] == 'start':
140 stream_sid = msg['start']['streamSid']
141 call_sid = msg['start']['callSid']
142
143 call_context = CallContext()
144
145 if os.getenv("RECORD_CALLS") == "true":
146 get_twilio_client().calls(call_sid).recordings.create({"recordingChannels": "dual"})
147
148 # Decide if the call the call was initiated from the UI or is an inbound
149 if call_sid not in call_contexts:
150 # Inbound call
151 call_context.system_message = os.environ.get("SYSTEM_MESSAGE")
152 call_context.initial_message = os.environ.get("INITIAL_MESSAGE")
153 call_context.call_sid = call_sid
154 call_contexts[call_sid] = call_context
155 else:
156 # Call from UI, reuse the existing context
157 call_context = call_contexts[call_sid]
158
159 llm_service.set_call_context(call_context)
160
161 stream_service.set_stream_sid(stream_sid)
162 transcription_service.set_stream_sid(stream_sid)
163
164 logger.info(f"Twilio -> Starting Media Stream for {stream_sid}")
165 await tts_service.generate({
166 "partialResponseIndex": None,
167 "partialResponse": call_context.initial_message
168 }, 1)
169 elif msg['event'] == 'media':
170 asyncio.create_task(process_media(msg))
171 elif msg['event'] == 'mark':
172 label = msg['mark']['name']
173 if label in marks:
174 marks.remove(label)
175 elif msg['event'] == 'stop':
176 logger.info(f"Twilio -> Media stream {stream_sid} ended.")
177 break
178 message_queue.task_done()
179
180 try:
181 listener_task = asyncio.create_task(websocket_listener())
182 processor_task = asyncio.create_task(message_processor())
183
184 await asyncio.gather(listener_task, processor_task)
185 except asyncio.CancelledError:
186 logger.info("Tasks cancelled")
187 finally:
188 await transcription_service.disconnect()
