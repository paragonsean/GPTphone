108 async def generate(self, llm_reply: Dict[str, Any], interaction_count: int):
109 """
110 Generates speech audio based on the given partial response.
111
112 Args:
113 llm_reply (Dict[str, Any]): The partial response received from the LLM service.
114 interaction_count (int): The count of interactions.
115
116 Returns:
117 None
118
119 Raises:
120 Exception: If an error occurs in the ElevenLabs TTS service.
121
122 """
123 partial_response_index, partial_response = llm_reply['partialResponseIndex'], llm_reply['partialResponse']
124
125 if not partial_response:
126 return
127
128 try:
129 output_format = "ulaw_8000"
130 url = f"https://api.elevenlabs.io/v1/text-to-speech/{self.voice_id}/stream"
131 headers = {
132 "xi-api-key": self.api_key,
133 "Content-Type": "application/json",
134 "Accept": "audio/wav"
135 }
136 params = {
137 "output_format": output_format,
138 "optimize_streaming_latency": 4
139 }
140 data = {
141 "model_id": self.model_id,
142 "text": partial_response
143 }
144
145 async with aiohttp.ClientSession() as session:
146 async with session.post(url, headers=headers, params=params, json=data) as response:
147 if response.status == 200:
148 audio_content = await response.read()
149 audio_base64 = base64.b64encode(audio_content).decode('utf-8')
150 await self.createEvent('speech', partial_response_index, audio_base64, partial_response,
151 interaction_count)
152 except Exception as err:
153 logger.error("Error occurred in ElevenLabs TTS service", exc_info=True)
154 logger.error(str(err))

162 async def generate(self, llm_reply, interaction_count):
163 """
164 Generate text-to-speech audio based on the given partial response.
165
166 Args:
167 llm_reply (dict): The partial response received from the LLM.
168 interaction_count (int): The count of interactions.
169
170 Returns:
171 None
172
173 Raises:
174 Exception: If there is an error in the TTS generation process.
175 """
176 partial_response_index = llm_reply['partialResponseIndex']
177 partial_response = llm_reply['partialResponse']
178
179 if not partial_response:
180 return
181
182 try:
183 source = {
184 "text": partial_response
185 }
186
187 options = {
188 "model": "aura-asteria-en",
189 "encoding": "mulaw",
190 "sample_rate": 8000
191 }
192
193 response = await self.clientclient.asyncspeak.v("1").stream(
194 source={"text": partial_response},
195 options=options
196 )
197
198 if response.stream:
199 audio_content = response.stream.getvalue()
200
201 # Convert audio to numpy array
202 audio_array = np.frombuffer(audio_content, dtype=np.uint8)
203
204 # Trim the first 10ms (80 samples at 8000Hz) to remove the initial noise
205 trim_samples = 80
206 trimmed_audio = audio_array[trim_samples:]
207
208 # Convert back to bytes
209 trimmed_audio_bytes = trimmed_audio.tobytes()
210
211 audio_base64 = base64.b64encode(trimmed_audio_bytes).decode('utf-8')
212 await self.createEvent('speech', partial_response_index, audio_base64, partial_response, interaction_count)
213 else:
214 logger.error("Error in TTS generation: No audio stream returned")
215
216 except Exception as e:
217 logger.error(f"Error in TTS generation: {str(e)}")
