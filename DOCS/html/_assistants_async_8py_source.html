215 def format_annotation(self, text):
216 citations = []
217 text_value = text.value
218 for index, annotation in enumerate(text.annotations):
219 text_value = text_value.replace(annotation.text, f" [{index}]")
220
221 if file_citation := getattr(annotation, "file_citation", None):
222 cited_file = client.files.retrieve(file_citation.file_id)
223 citations.append(
224 f"[{index}] {file_citation.quote} from {cited_file.filename}"
225 )
226 elif file_path := getattr(annotation, "file_path", None):
227 link_tag = self.create_file_link(
228 annotation.text.split("/")[-1],
229 file_path.file_id,
230 )
231 text_value = re.sub(r"\[(.*?)\]\s*\‍(\s*(.*?)\s*\‍)", link_tag, text_value)
232 text_value += "\n\n" + "\n".join(citations)
