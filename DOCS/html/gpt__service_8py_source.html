75 # Split the text into sentences, keeping the separators
76 sentences = re.split(r'([.!?])', text)
77 # Pair the sentences with their separators
78 sentences = [''.join(sentences[i:i + 2]) for i in range(0, len(sentences), 2)]
79 return sentences

83 self.sentence_buffer += text
84 sentences = self.split_into_sentences(self.sentence_buffer)

92 self.partial_response_index += 1

95 self.sentence_buffer = sentences[-1] if sentences else ""
